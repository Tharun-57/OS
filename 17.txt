#include<stdio.h>
int main(){
int n,i,head,move=0;
int req[20];
printf("Enter number of disk requests: ");
scanf("%d",&n);
printf("Enter disk request sequence:\n");
for(i=0;i<n;i++)scanf("%d",&req[i]);
printf("Enter initial head position: ");
scanf("%d",&head);
for(i=0;i<n;i++){
move+=abs(req[i]-head);
head=req[i];
}
printf("Total head movement = %d\n",move);
return 0;
}
