#include<stdio.h>
int main(){
int u,f,i,j;
char user[10][20],file[10][10][20];
printf("Enter number of users: ");
scanf("%d",&u);
for(i=0;i<u;i++){
printf("Enter user name %d: ",i+1);
scanf("%s",user[i]);
printf("Enter number of files for %s: ",user[i]);
scanf("%d",&f);
for(j=0;j<f;j++){
printf("Enter file name: ");
scanf("%s",file[i][j]);
}
printf("User: %s\nFiles:\n",user[i]);
for(j=0;j<f;j++)printf("%s\n",file[i][j]);
}
return 0;
}
