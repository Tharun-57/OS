#include<stdio.h>
#include<pthread.h>
void*run(void*a){
int x=*(int*)a;
printf("Thread %d is executing\n",x);
return NULL;
}
int main(){
pthread_t t1,t2;
int a=1,b=2;
pthread_create(&t1,NULL,run,&a);
pthread_create(&t2,NULL,run,&b);
pthread_join(t1,NULL);
pthread_join(t2,NULL);
printf("Both threads completed\n");
return 0;
}
