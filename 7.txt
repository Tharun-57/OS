#include <stdio.h>
#include <unistd.h>
#include <string.h>

int main() {
    int fd[2];
    char msg[20];

    pipe(fd);

    if (fork() == 0) {
        close(fd[0]);
        strcpy(msg, "Hello");
        write(fd[1], msg, strlen(msg) + 1);
    } else {
        close(fd[1]);
        read(fd[0], msg, sizeof(msg));
        printf("Message received: %s\n", msg);
    }

    return 0;
}
